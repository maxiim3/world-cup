import {TeamModel} from "./TeamModel"
type TeamMatchProps = {
   props : TeamModel,
   score: number
}
class Match {
   team1 : TeamMatchProps
   team2 : TeamMatchProps

   constructor(team1:TeamModel, team2:TeamModel) {
      this.team1 = {
         props: team1,
         score: 0
      };
      this.team2 = {
         props: team2,
         score: 0
      };
   }

   updateScore(team:TeamMatchProps, n:number) {
      const getScore = team.score
      return team.score = getScore + n
   }

   calculateScore(team:TeamMatchProps) :number {
      return Math.floor((Math.random() * 3) + team.props.xp / 100)
   }

   updateTeam(resT1:number, resT2:number) {
      const balance = (Math.abs(resT1 - resT2))
      if (resT1 > resT2) {
         this.team1.props.xp += balance
         this.team2.props.xp -= balance
      }
      else if (resT1 < resT2) {
         this.team2.props.xp += balance
         this.team1.props.xp -= balance
      }
   }

   round(scoreTeam1:number, scoreTeam2:number) :Object {
      scoreTeam1 += this.calculateScore(this.team1)
      scoreTeam2 += this.calculateScore(this.team2)

      return {scoreTeam1, scoreTeam2}
   }
